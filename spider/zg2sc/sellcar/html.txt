$(document).ready(function() {	
	//感兴趣的二手车
	$.post("/usedcar/getInterested.do", function(data) {
		var jsonObj = eval("(" + data + ")");
		var res = jsonObj.success;
		var content = "";
		if(res){
			$("#search_car_gxq_dt_div").css("display","block");
			//$("#search_car_gxq_jt").css("display","none");
			var carVo = jsonObj.carVo;
			for(var i=0;i<carVo.length;i++){
				var car = carVo[i];
				content += '<li><span><a href="'+car.carurl+'" target="_blank">';
				if(null == car.smphoto || car.smphoto == ''){
					content += '<img src="http://img.zg2sc.cn/img2013/nocarpo.jpg" />';
				} else {
					content += '<img src="http://www.zg2sc.cn'+car.smphoto+'" />';
				}
				content += '</a></span><p class="mc"><a>'+car.regtime+car.car_class;
				content += '</a></p><p class="jg"><a>'+car.city;
				content += '<strong>'+car.price+'万元</strong></a></p></li>';
			}
			$("#search_car_gxq_dt").html(content);
		}
	});
	
	//品牌关注度
	$.post("/usedcar/getBrandRanking.do", function(data) {
		var jsonObj = eval("(" + data + ")");
		var res = jsonObj.success;
		var content = "";
		if(res){
			$("#search_gzd_dt").css("display","block");
			$("#search_gzd_jt").css("display","none");
			var brVo = jsonObj.brVo;
			for(var i=0;i<brVo.length;i++){
				var br = brVo[i];
				content += '<dt><span class="a1">' + (i+1) + '</span>' + br.brandName;
				if(br.seqfloat > 0){
					content += '</dt><dd><span class="a1">'+br.seqfloat;
				} else if(br.seqfloat < 0) {
					content += '</dt><dd><span class="a2">'+br.seqfloat;
				} else {
					content += '</dt><dd><span>-';
				}
				content += '</span></dd><div class="clear">&nbsp;</div>';
			}
			$("#search_gzd_dt").html(content);
		}
	});
	
});
